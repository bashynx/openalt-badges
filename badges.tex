\newcommand{\badgeO}[3]{\ticket{
  \put(0,0){\includegraphics[width=109mm]{#1}}
  \put(54.5,32){\makebox[0mm]{\parbox[t][][t]{90mm}{\fontsize{30}{36}\centering\rhDisplay\bfseries #2}}} % name
  \put(54.5,18){\makebox[0mm]{\parbox[t][][t]{90mm}{\fontsize{30}{36}\centering\rhDisplay\bfseries #3}}} % surname
}}
